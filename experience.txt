                // Add click event to timeline navigation items
                timelineNavItems.forEach(item => {

                        e.preventDefault();
                        
                        // Remove active class from all items
                        timelineNavItems.forEach(navItem => {
                            navItem.classList.remove('bg-gradient-to-r', 'from-primary/10', 'to-accent/10');

                        // Add active class to clicked item
                        this.classList.add('bg-gradient-to-r', 'from-primary/10', 'to-accent/10');
                        
                        // Smooth scroll to target with improved positioning
                        const targetId = this.getAttribute('href');
                        const targetElement = document.querySelector(targetId);
                        
                        if (targetElement) {
                            // Add a small delay to ensure proper calculation
                            setTimeout(() => {
                                const targetPosition = targetElement.getBoundingClientRect().top + window.pageYOffset - headerHeight;
                                window.scrollTo({
                                    top: targetPosition,
                                    behavior: 'smooth'
                                });
                            }, 50);
                        }
                    });
                });
                
                // Add scroll spy functionality
                window.addEventListener('scroll', function() {
                    const scrollPosition = window.scrollY + headerHeight + 50;
                    
                    // Check each experience item's position
                    document.querySelectorAll('[id^="exp-"]').forEach(section => {
                        const sectionTop = section.offsetTop;
                        const sectionHeight = section.offsetHeight;
                        
                        if (scrollPosition >= sectionTop && scrollPosition < sectionTop + sectionHeight) {

                                correspondingNavItem.classList.add('bg-gradient-to-r', 'from-primary/10', 'to-accent/10');

                        }
